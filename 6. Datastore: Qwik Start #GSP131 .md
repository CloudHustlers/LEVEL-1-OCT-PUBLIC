from google.cloud import datastore
import datetime
client = datastore.Client()
key = client.key("Task")
entity = datastore.Entity(key=key)
entity["description"] = "Learn Google Cloud Datastore"
entity["created"] = datetime.datetime.now()  
entity["done"] = False
client.put(entity)
EOF
python s.py
